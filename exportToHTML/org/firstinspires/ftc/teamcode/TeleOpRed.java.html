104      @Override
105      public void loop(){
106          //movement
107  
108  
109          forward = gamepad1.left_stick_y;
110          strafe = (-gamepad1.left_stick_x);
111          rotate = -gamepad1.right_stick_x;
112          if(gamepad1.right_trigger > 0.2 || gamepad1.left_trigger > 0.2){
113              forward = forward*0.25;
114              strafe = strafe*0.25;
115              rotate = rotate*0.25;
116          }
117          getLLRotationOffset();
118          double rot = 0;
119          LLResult llResult = limelight3A.getLatestResult();
120          if(llResult != null & llResult.isValid()){
121              rot = llResult.getTx();
122          }else{
123              rot = -1;
124          }
125  
126          //line up with target
127          if(gamepad1.right_bumper) { //TARGET
128              //fire line up
129  
130  
131  
132              if(rot == -1){//cant see the tag or other problem
133  
134                      drive.drive(0,0,-targetSpeedHigh);
135  
136              }else{//tag visible
137                  if (rot >= 20){drive.drive(0,strafe,targetSpeedMed);}
138                  else if(rot < 20 & (rot > (2+shootOfsetValue))){ drive.drive(0,strafe,-targetSpeedLow);}
139                  else if((rot <=(2+shootOfsetValue))  & (rot >= (-2+shootOfsetValue))){ drive.drive(0,strafe,0);}
140                  else if(rot > -20 & (rot < (-2+shootOfsetValue))){ drive.drive(0,strafe,targetSpeedLow);}
141                  //else if(rot <= -20){drive.drive(0,0,-targetSpeedMed);}
142                  else{drive.drive(0,strafe,0);}//catchall
143              }
144          }else if(gamepad1.left_bumper && (Math.abs(rot) <= 2)){
145              //fire line up
146  
147              if(rot == -1){//cant see the tag or other problem
148  
149                  drive.drive(0,0,targetSpeedHigh);
150  
151              }else{//tag visible
152                  //if (rot >= 20){drive.drive(0,0,-targetSpeedMed);}
153                  /*else*/ if(rot <= -20){drive.drive(0,strafe,targetSpeedMed);}
154                  else if(rot > -20 & rot < -2){ drive.drive(0,strafe,targetSpeedLow);}
155                  else if(rot <=2  & rot >= -2){ drive.drive(0,strafe,0);}
156                  else if(rot < 20 & rot > 2){ drive.drive(0,strafe,-targetSpeedLow);}//you can strafe arround the target
157                  else{drive.drive(0,strafe,0);}//catchall
158              }
159          }else{
160              drive.drive(forward,strafe,rotate);
161          }
162          double distance = getLLDistance();
163          int TargetVelocity;
164          if(distance > 130){
165              TargetVelocity = (int) (idleSpeed + ((distance-165)*1));//was 860- down town shooting
166          }else if(distance < 130 & distance > 60){
167              TargetVelocity = (int) (idleSpeed + ((distance - 60)*0.74)); //set the intermediate power orignal 580 dist 70 was .66
168          }else if(distance < 60){
169              TargetVelocity = idleSpeed;//original 580
170          }else{
171              TargetVelocity = idleSpeed;//original 630
172          }
173          if(distance >= 130){//calibrating for distance
174              shootOfsetValue = -3;
175          }else{
176              shootOfsetValue = 0;
177          }
178          if(gamepad2.x){
179              idleSpeed = 620;
180          }
181          if(gamepad2.b){
182              idleSpeed = 740;
183          }
184          double ltdef = 0;
185          double rtdef = 0;
186          double velocity = shooterMotor.getVelocity();
187          if(gamepad2.left_bumper || gamepad2.right_bumper) {
188              rtIntake.setPower(-1);
189          }else if(gamepad2.left_stick_y > 0.5 || gamepad2.right_stick_y > 0.5){// joysticks move intake
190              rtIntake.setPower(-1);
191              if(color.getDistance(DistanceUnit.CM) > 3.6){
192                  ltdef= 0.25;//was .25
193              }else{
194                  ltdef = 0;
195              }
196              if(rtcolor.getDistance(DistanceUnit.CM) > 3.6){
197                  rtdef = 0.25;//was.25
198  
199              }else{
200                  rtdef = 0;
201              }
202  
203          }else if(gamepad2.y){
204              rtIntake.setPower(1);
205          }else{
206              rtIntake.setPower(0);
207          }
208          if(gamepad2.left_bumper && rot < 2 && rot > -2 && (!(rot == -1)) && shooterMotor2.getVelocity() > (TargetVelocity - 30)){
209  
210              ballStopLeft.setPosition(0.7);
211          }else{
212              ballStopLeft.setPosition(0.25);
213          }
214          if(gamepad2.right_bumper && rot < 2 && rot > -2 && (!(rot == -1)) && shooterMotor.getVelocity() > (TargetVelocity - 30)){
215  
216              ballStopRight.setPosition(0.7);
217          }else{
218              ballStopRight.setPosition(0.3);
219          }
220          if(rot < 2 && rot > -2 && (!(rot == -1))){//dont shoot unless within zone
221              if(gamepad2.left_bumper ){
222                  ltFire.setPower(1);
223              }else{
224                  ltFire.setPower(ltdef);
225              }
226              if(gamepad2.right_bumper ){
227                  rtFire.setPower(-1);
228              }else{
229                  rtFire.setPower(-rtdef);
230              }
231          }else{
232              ltFire.setPower(ltdef);
233              rtFire.setPower(-rtdef);
234          }
235  
236  
237          if(velocity < TargetVelocity){//speed up /!\ shooter speed adjustments /!\ SHOOT
238              shooterMotor.setPower(1);
239              //led2.setState(true);//off leds
240              led3.setState(true);
241          }else {//fast eneough
242              shooterMotor.setPower(0.5);
243              //led2.setState(false);
244              led3.setState(false);
245          }
246          if(shooterMotor2.getVelocity() < TargetVelocity){//speed up /!\ shooter speed adjustments /!\ SHOOT
247              shooterMotor2.setPower(1);
248              led0.setState(true);
249              //led1.setState(true);
250  
251          }else {//fast eneough
252              shooterMotor2.setPower(0.5);
253              led0.setState(false);
254              //led1.setState(false);
255  
256          }
257          if(color.getDistance(DistanceUnit.CM) > 3.4){
258              led1.setState(true);
259          }else{
260              led1.setState(false);
261          }
262          if(rtcolor.getDistance(DistanceUnit.CM) > 3.4){
263              led2.setState(true);
264          }else{
265              led2.setState(false);
266          }
267  
268  
269          telemetry.addData("power:",setSpeed);
270          telemetry.addData("speed:",velocity);
271          telemetry.addData("speed2:",shooterMotor2.getVelocity());
272          telemetry.addData("target",TargetVelocity);
273  
274

282      private double getLLRotationOffset(){
283          LLResult llResult = limelight3A.getLatestResult();
284          if (llResult != null & llResult.isValid()) {
285              telemetry.addData("target X offset", llResult.getTx());
286              telemetry.addData("Target y offset", llResult.getTy());
287              telemetry.addData("Target area offset", llResult.getTa());
288              double y = llResult.getTy();
289              double angleRadians = 3.14*((23+y)/180);
290              double targetDist = 26.25 / tan(angleRadians);
291              telemetry.addData("distance:",targetDist);
292              return llResult.getTx();
293          }else{
294              return -1;
295          }
296  
297      }
298      private double getLLDistance(){
299          LLResult llResult = limelight3A.getLatestResult();
300          if (llResult != null & llResult.isValid()) {
301              telemetry.addData("target X offset", llResult.getTx());
302              telemetry.addData("Target y offset", llResult.getTy());
303              telemetry.addData("Target area offset", llResult.getTa());
304              double y = llResult.getTy();
305              double angleRadians = 3.14*((18+y)/180);
306              double targetDist = 22.5 / tan(angleRadians);
307              telemetry.addData("distance:",targetDist);
308              return targetDist;
309          }else{
310              return -1;
311          }
